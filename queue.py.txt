class Queue:
	def __init__(self):
		self.items=[]
	def enque(self,item):
		self.items.insert(0,item)
	def deque(self):
		return self.items.pop()
	def size(self):
		return len(self.items)
	def isempty(self):
		return self.items==[]
q=Queue()
q.enque(2)
q.enque(3)

print(q.items)
print(q.deque())

print(q.items)
print(q.size())
print(q.isempty())
